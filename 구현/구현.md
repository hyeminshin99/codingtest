#### 예제4-1) 상하좌우문제
```python
N = int(input())
x, y = 1, 1
plans = input().split()

# L, R, U, D
dx = [0,0,-1,1] #일반적인 좌표와 다른 점 주의!
dy = [-1,1,0,0]
move_types = ['L', 'R', 'U', 'D']

for plan in plans:
    for i in range(4):
        if plan == move_types[i]:
            if x+dx[i]>=1 and y+dy[i]>=1 and x+dx[i]<=N and y+dy[i]<=N:
                x += dx[i]
                y += dy[i]
                break
print(x, y)

```
